**ECS 核心概念**：

  

```typescript

// Entity: 唯一 ID，只是一个容器

type EntityId = number;

  

// Component: 纯数据，无逻辑

interface PositionComponent {

  x: number;

  y: number;

}

  

interface VelocityComponent {

  vx: number;

  vy: number;

}

  

// System: 处理特定组件组合的逻辑

class MovementSystem {

  update(entities: Entity[], deltaTime: number) {

    for (const entity of entities) {

      if (entity.has(PositionComponent) && entity.has(VelocityComponent)) {

        const pos = entity.get(PositionComponent);

        const vel = entity.get(VelocityComponent);

        pos.x += vel.vx * deltaTime;

        pos.y += vel.vy * deltaTime;

      }

    }

  }

}

```

```typescript
// 项目 2.4.1: 实现四叉树空间索引

class QuadTree {

  private bounds: AABB;

  private capacity: number;

  private entities: Entity[] = [];

  private divided: boolean = false;

  private children: QuadTree[] = [];

  

  insert(entity: Entity): boolean {

    if (!this.bounds.contains(entity.position)) {

      return false;

    }

  

    if (this.entities.length < this.capacity) {

      this.entities.push(entity);

      return true;

    }

  

    if (!this.divided) {

      this.subdivide();

    }

  

    for (const child of this.children) {

      if (child.insert(entity)) {

        return true;

      }

    }

  

    return false;

  }

  

  query(range: AABB): Entity[] {

    const found: Entity[] = [];

  

    if (!this.bounds.intersects(range)) {

      return found;

    }

  

    for (const entity of this.entities) {

      if (range.contains(entity.position)) {

        found.push(entity);

      }

    }

  

    if (this.divided) {

      for (const child of this.children) {

        found.push(...child.query(range));

      }

    }

  

    return found;

  }

}

  

// 项目 2.4.2: 实现视口剔除

class CullingSystem {

  update(camera: Camera, entities: Entity[]) {

    const frustum = camera.getFrustum();

    const visibleEntities = this.quadTree.query(frustum);
```

#### 算法 1: 手绘线条 (Sketchy Line)

  

```typescript

// Rough.js 的线条算法

function sketchyLine(from: Point, to: Point, roughness: number): Point[] {

  const length = distance(from, to);

  const segmentCount = Math.ceil(length / 10); // 每 10px 一个段

  const points: Point[] = [from];

  

  for (let i = 1; i < segmentCount; i++) {

    const t = i / segmentCount;

    const basePoint = lerp(from, to, t);

    // 添加随机偏移

    const offset = (Math.random() - 0.5) * roughness;

    const perpendicular = normalize(perpendicular(to - from));

    const point = basePoint + perpendicular * offset;

    points.push(point);

  }

  

  points.push(to);

  return points;

}

```

  

#### 算法 2: Hachure 填充 (交叉线填充)

  

```typescript

function hachureFill(polygon: Point[], angle: number, gap: number): Line[] {

  const lines: Line[] = [];

  const bounds = getBounds(polygon);

  // 生成平行线

  const step = gap;

  const diagonal = Math.sqrt(bounds.width ** 2 + bounds.height ** 2);

  const lineCount = Math.ceil(diagonal / step);

  

  for (let i = 0; i < lineCount; i++) {

    const y = bounds.top + i * step;

    const line = {

      from: { x: bounds.left, y },

      to: { x: bounds.right, y },

    };

  

    // 旋转线条

    const rotatedLine = rotateLine(line, angle, bounds.center);

  

    // 裁剪到多边形内

    const clippedLine = clipLineToPolygon(rotatedLine, polygon);

    if (clippedLine) {

      lines.push(clippedLine);

    }

  }

  

  return lines;

}

```

```glsl
// Vertex Shader

attribute vec2 aPosition;

attribute vec2 aUV;

attribute float aSegmentIndex; // 线段索引

  

uniform mat3 uProjectionMatrix;

uniform mat3 uWorldTransform;

uniform float uRoughness;

uniform float uSeed;

  

varying vec2 vUV;

varying float vAlpha;

  

// 伪随机函数

float random(vec2 st) {

  return fract(sin(dot(st.xy, vec2(12.9898, 78.233))) * 43758.5453123);

}

  

void main() {

  vec2 position = aPosition;

  // 添加随机偏移

  float offset = (random(vec2(aSegmentIndex, uSeed)) - 0.5) * uRoughness;

  position += vec2(-offset, offset); // 垂直于线条方向

  vec3 worldPos = uWorldTransform * vec3(position, 1.0);

  gl_Position = vec4((uProjectionMatrix * worldPos).xy, 0.0, 1.0);

  vUV = aUV;

  vAlpha = 1.0;

}

  

// Fragment Shader

precision mediump float;

  

varying vec2 vUV;

varying float vAlpha;

  

uniform vec4 uColor;

  

void main() {

  gl_FragColor = vec4(uColor.rgb, uColor.a * vAlpha);
```

#### Shader 2: 手绘矩形 Shader (SDF + 噪声)

  

```glsl

// Fragment Shader

precision mediump float;

  

varying vec2 vUV;

  

uniform vec4 uColor;

uniform vec2 uSize;

uniform float uRoughness;

uniform float uSeed;

  

// SDF 矩形

float sdBox(vec2 p, vec2 b) {

  vec2 d = abs(p) - b;

  return length(max(d, 0.0)) + min(max(d.x, d.y), 0.0);

}

  

// Perlin 噪声（简化版）

float noise(vec2 st) {

  return fract(sin(dot(st.xy, vec2(12.9898, 78.233))) * 43758.5453123);

}

  

void main() {

  vec2 uv = vUV * 2.0 - 1.0;

  // 添加噪声扰动

  float n = noise(uv * 10.0 + uSeed) * uRoughness;

  uv += vec2(n, n);

  // SDF 矩形

  float d = sdBox(uv, uSize * 0.5);

  // 描边效果

  float strokeWidth = 0.02;

  float stroke = smoothstep(strokeWidth, 0.0, abs(d));

  // 多次描边（通过噪声模拟）

  float stroke2 = smoothstep(strokeWidth + 0.01, 0.0, abs(d + n * 0.1));

  float finalStroke = max(stroke, stroke2 * 0.5);

  gl_FragColor = vec4(uColor.rgb, finalStroke);

}

```

  

#### Shader 3: Hachure 填充 Shader

  

```glsl

// Fragment Shader

precision mediump float;

  

varying vec2 vUV;

  

uniform vec4 uColor;

uniform float uHachureAngle;

uniform float uHachureGap;

uniform float uRoughness;

  

float noise(vec2 st) {

  return fract(sin(dot(st.xy, vec2(12.9898, 78.233))) * 43758.5453123);

}

  

void main() {

  vec2 uv = vUV;

  // 旋转 UV

  float angle = uHachureAngle;

  mat2 rotation = mat2(cos(angle), -sin(angle), sin(angle), cos(angle));

  vec2 rotatedUV = rotation * (uv - 0.5) + 0.5;

  // 生成平行线

  float line = fract(rotatedUV.y / uHachureGap);

  // 添加噪声

  float n = noise(uv * 50.0) * uRoughness;

  line += n;
```

**核心概念**：

  

```typescript

// 时间轴 (Timeline)

class Timeline {

  private tracks: Track[] = [];

  private currentTime: number = 0;

  private duration: number = 0;

  private playing: boolean = false;

  

  addTrack(track: Track) {

    this.tracks.push(track);

    this.duration = Math.max(this.duration, track.endTime);

  }

  

  play() {

    this.playing = true;

  }

  

  pause() {

    this.playing = false;

  }

  

  seek(time: number) {

    this.currentTime = time;

    this.update(0);

  }

  

  update(deltaTime: number) {

    if (!this.playing) return;

  

    this.currentTime += deltaTime;

  

    if (this.currentTime > this.duration) {

      this.currentTime = 0; // 循环播放

    }

  

    for (const track of this.tracks) {

      track.update(this.currentTime);

    }

  }

}

  

// 轨道 (Track)

interface Track {

  startTime: number;

  endTime: number;

  target: any; // 动画目标（实体、相机等）

  update(time: number): void;

}

  

// 关键帧 (Keyframe)

interface Keyframe<T> {

  time: number;

  value: T;

  easing: EasingFunction;

}

  

// 属性轨道 (Property Track)

class PropertyTrack<T> implements Track {

  startTime: number;

  endTime: number;

  target: any;

  property: string;

  keyframes: Keyframe<T>[];

  

  constructor(target: any, property: string, keyframes: Keyframe<T>[]) {

    this.target = target;

    this.property = property;

    this.keyframes = keyframes.sort((a, b) => a.time - b.time);

    this.startTime = keyframes[0].time;

    this.endTime = keyframes[keyframes.length - 1].time;

  }

  

  update(time: number) {

    if (time < this.startTime || time > this.endTime) return;

  

    // 找到当前时间所在的关键帧区间

    let i = 0;

    while (i < this.keyframes.length - 1 && this.keyframes[i + 1].time < time) {

      i++;

    }

  

    const keyframe1 = this.keyframes[i];

    const keyframe2 = this.keyframes[i + 1];

  

    if (!keyframe2) {

      this.target[this.property] = keyframe1.value;

      return;

    }

  

    // 计算插值

    const t = (time - keyframe1.time) / (keyframe2.time - keyframe1.time);

    const easedT = keyframe1.easing(t);

    const value = this.interpolate(keyframe1.value, keyframe2.value, easedT);

  

    this.target[this.property] = value;

  }

  

  private interpolate(a: T, b: T, t: number): T {

    // 根据类型进行插值

    if (typeof a === 'number' && typeof b === 'number') {

      return (a + (b - a) * t) as T;

    }

    // ... 其他类型的插值

  }

}

```

```typescript

class PlaybackSystem {

  private events: RecordedEvent[];

  private currentIndex: number = 0;

  private currentTime: number = 0;

  private playing: boolean = false;

  

  constructor(events: RecordedEvent[]) {

    this.events = events;

  }

  

  play() {

    this.playing = true;

    this.currentTime = 0;

    this.currentIndex = 0;

  }

  

  pause() {

    this.playing = false;

  }

  

  seek(time: number) {

    this.currentTime = time;

    // 找到对应的事件索引

    this.currentIndex = 0;

    while (

      this.currentIndex < this.events.length &&

      this.events[this.currentIndex].time < time

    ) {

      this.currentIndex++;

    }

  }

  

  update(deltaTime: number, world: World) {

    if (!this.playing) return;

  

    this.currentTime += deltaTime * 1000; // 转换为毫秒

  

    // 执行所有应该在当前时间触发的事件

    while (

      this.currentIndex < this.events.length &&

      this.events[this.currentIndex].time <= this.currentTime

    ) {

      const event = this.events[this.currentIndex];

      this.executeEvent(event, world);

      this.currentIndex++;

    }

  

    // 回放结束

    if (this.currentIndex >= this.events.length) {

      this.playing = false;

    }

  }

  

  private executeEvent(event: RecordedEvent, world: World) {

    switch (event.type) {

      case 'entity:created':

        this.createEntity(event.data, world);

        break;

      case 'component:changed':

        this.updateComponent(event.data, world);

        break;

      case 'camera:changed':

        this.updateCamera(event.data);

        break;

    }

  }

  

  private createEntity(data: any, world: World) {

    const entity = addEntity(world);

    for (const [componentName, componentData] of Object.entries(data.components)) {

      const component = getComponentByName(componentName);

      addComponent(world, component, entity);

      for (const [key, value] of Object.entries(componentData)) {

        component[key][entity] = value;

      }

    }

  }

  

  private updateComponent(data: any, world: World) {

    const entity = data.id;

    const component = getComponentByName(data.component);

    for (const [key, value] of Object.entries(data.value)) {

      component[key][entity] = value;

    }

  }

  

  private updateCamera(data: any) {

    this.camera.x = data.x;

    this.camera.y = data.y;

    this.camera.zoom = data.zoom;

  }

}
```

```typescript
class VideoExporter {

  private canvas: HTMLCanvasElement;

  private mediaRecorder: MediaRecorder;

  private chunks: Blob[] = [];

  

  constructor(canvas: HTMLCanvasElement) {

    this.canvas = canvas;

  }

  

  async startRecording(options: { fps?: number; videoBitsPerSecond?: number } = {}) {

    const stream = this.canvas.captureStream(options.fps || 60);

    this.mediaRecorder = new MediaRecorder(stream, {

      mimeType: 'video/webm;codecs=vp9',

      videoBitsPerSecond: options.videoBitsPerSecond || 5000000,

    });

  

    this.mediaRecorder.ondataavailable = (e) => {

      if (e.data.size > 0) {

        this.chunks.push(e.data);

      }

    };

  

    this.mediaRecorder.start();

  }

  

  stopRecording(): Promise<Blob> {

    return new Promise((resolve) => {

      this.mediaRecorder.onstop = () => {

        const blob = new Blob(this.chunks, { type: 'video/webm' });

        this.chunks = [];

        resolve(blob);

      };

  

      this.mediaRecorder.stop();

    });

  }

  

  async exportVideo(playbackSystem: PlaybackSystem): Promise<Blob> {

    await this.startRecording();

    playbackSystem.play();

  

    // 等待回放完成

    await new Promise((resolve) => {

      const checkInterval = setInterval(() => {

        if (!playbackSystem.playing) {

          clearInterval(checkInterval);

          resolve(null);

        }

      }, 100);

    });

  

    return await this.stopRecording();

  }

  

  downloadVideo(blob: Blob, filename: string = 'whiteboard-animation.webm') {

    const url = URL.createObjectURL(blob);

    const a = document.createElement('a');

    a.href = url;

    a.download = filename;

    a.click();

    URL.revokeObjectURL(url);

  }

}
```

```typescript
      this.ydoc

  private setupSync() {

    // 监听远程变化

    this.yEntities.observe((event) => {

      event.changes.keys.forEach((change, key) => {

        if (change.action === 'add' || change.action === 'update') {

          const data = this.yEntities.get(key);

          this.applyRemoteChange(key, data);

        } else if (change.action === 'delete') {

          this.deleteEntity(key);

        }

      });

    });

  }

  

  syncEntity(entity: number, world: World) {

    const data = this.serializeEntity(entity, world);

    this.yEntities.set(entity.toString(), data);

  }

  

  private serializeEntity(entity: number, world: World): any {

    const data: any = { id: entity };

  

    if (hasComponent(world, Transform, entity)) {

      data.transform = {

        x: Transform.x[entity],

        y: Transform.y[entity],

        scaleX: Transform.scaleX[entity],

        scaleY: Transform.scaleY[entity],

        rotation: Transform.rotation[entity],

      };

    }

  

    // ... 序列化其他组件

  

    return data;

  }

  

  private applyRemoteChange(key: string, data: any) {

    if (!entityExists(this.world, entity)) {
```
